-- Skills table structure and sample data
-- This represents different hacking and programming skills

CREATE TABLE skills (
    id INTEGER PRIMARY KEY,
    skill_name TEXT NOT NULL,
    category TEXT,
    difficulty TEXT,
    points INTEGER,
    description TEXT
);
