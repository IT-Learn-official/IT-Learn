-- Projects table structure and sample data
-- This represents various hacking projects and challenges

CREATE TABLE projects (
    id INTEGER PRIMARY KEY,
    project_name TEXT NOT NULL,
    description TEXT,
    difficulty TEXT,
    status TEXT,
    points_reward INTEGER,
    deadline TEXT
);
